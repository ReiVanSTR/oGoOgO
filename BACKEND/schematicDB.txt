[[Creating main database]]

CREATE TABLE users (
	ID int NOT NULL AUTO_INCREMENT,
	user VARCHAR(255) NOT NULL,
	status BOOLEAN DEFAULT FALSE, 
	PRIMARY KEY(ID)
);

[[Updating users status]]
UPDATE users
SET status = status
WHERE user = user;

[[Create new user]]
INSERT INTO users (user, status)
VALUES (user, False);

[[Del user]]
DELETE FROM users WHERE user = user (username);
